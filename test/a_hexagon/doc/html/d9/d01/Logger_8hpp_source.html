71 static void init(const std::string& app_name = "a_hexagon",
72 std::size_t queue_size = 8192,
73 std::size_t thread_count = 1) {
74
75 // Initialize async thread pool
76 spdlog::init_thread_pool(queue_size, thread_count);
77
78 // Create console sink with colors
79 auto console_sink = std::make_shared<spdlog::sinks::stdout_color_sink_mt>();
80 console_sink->set_level(spdlog::level::trace);
81 console_sink->set_pattern("[%Y-%m-%d %H:%M:%S.%e] [%^%l%$] [%n] %v");
82
83 // Create async logger with non-blocking overflow policy
84 auto logger = std::make_shared<spdlog::async_logger>(
85 app_name,
86 console_sink,
87 spdlog::thread_pool(),
88 spdlog::async_overflow_policy::overrun_oldest // Non-blocking for real-time
89 );
90
91 // Set as default logger
92 spdlog::set_default_logger(logger);
93 spdlog::set_level(spdlog::level::info);
94
95 // Flush on error level and above
96 spdlog::flush_on(spdlog::level::err);
97
98 initialized_ = true;
99 }

108 static void initWithFile(const std::string& app_name,
109 const std::string& log_file,
110 std::size_t max_file_size = 5 * 1024 * 1024,
111 std::size_t max_files = 3) {
112
113 spdlog::init_thread_pool(8192, 1);
114
115 // Console sink
116 auto console_sink = std::make_shared<spdlog::sinks::stdout_color_sink_mt>();
117 console_sink->set_level(spdlog::level::info);
118 console_sink->set_pattern("[%Y-%m-%d %H:%M:%S.%e] [%^%l%$] [%n] %v");
119
120 // Rotating file sink
121 auto file_sink = std::make_shared<spdlog::sinks::rotating_file_sink_mt>(
122 log_file, max_file_size, max_files);
123 file_sink->set_level(spdlog::level::trace);
124 file_sink->set_pattern("[%Y-%m-%d %H:%M:%S.%e] [%l] [%n] [%t] %v");
125
126 // Multi-sink async logger
127 std::vector<spdlog::sink_ptr> sinks{console_sink, file_sink};
128 auto logger = std::make_shared<spdlog::async_logger>(
129 app_name,
130 sinks.begin(), sinks.end(),
131 spdlog::thread_pool(),
132 spdlog::async_overflow_policy::overrun_oldest
133 );
134
135 spdlog::set_default_logger(logger);
136 spdlog::set_level(spdlog::level::info);
137 spdlog::flush_on(spdlog::level::err);
138
139 initialized_ = true;
140 }
