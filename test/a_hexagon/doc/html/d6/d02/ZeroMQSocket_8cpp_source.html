81bool ZeroMQSocket::connect(const std::string& endpoint, ConnectionMode mode) {
82 std::lock_guard<std::mutex> lock(mutex_);
83
84 if (connected_.load()) {
85 LOG_WARN("ZeroMQSocket already connected to: {}", endpoint_);
86 return true;
87 }
88
89 try {
90 socket_ = std::make_unique<zmq::socket_t>(*context_, socketType_);
91
92 // Set common socket options
93 socket_->set(zmq::sockopt::linger, 0);
94
95 // Connect or Bind based on mode
96 if (mode == ConnectionMode::Connect) {
97 socket_->connect(endpoint);
98 LOG_INFO("ZeroMQ socket connected to: {}", endpoint);
99 } else {
100 socket_->bind(endpoint);
101 LOG_INFO("ZeroMQ socket bound to: {}", endpoint);
102 }
103
104 endpoint_ = endpoint;
105 mode_ = mode;
106 connected_.store(true);
107 return true;
108
109 } catch (const zmq::error_t& e) {
110 LOG_ERROR("ZeroMQ connect/bind failed: {} - {}", endpoint, e.what());
111 socket_.reset();
112 return false;
113 }

120bool ZeroMQSocket::send(const std::vector<uint8_t>& data, const std::string& group) {
121 if (!connected_.load()) {
122 LOG_WARN("Cannot send - socket not connected");
123 return false;
124 }
125
126 std::lock_guard<std::mutex> lock(mutex_);
127
128 if (!socket_) {
129 LOG_WARN("Cannot send - socket is null");
130 return false;
131 }
132
133 try {
134 zmq::message_t message(data.size());
135 std::memcpy(message.data(), data.data(), data.size());
136
137 // RADIO socket requires set_group for UDP multicast (Draft API)
138 if (!group.empty() && (socketType_ == ZMQ_RADIO)) {
139 message.set_group(group.c_str());
140 LOG_TRACE("Set RADIO message group: {}", group);
141 }
142
143 auto result = socket_->send(message, zmq::send_flags::dontwait);
144
145 if (!result.has_value()) {
146 LOG_WARN("ZeroMQ send returned no result");
147 return false;
148 }
149
150 LOG_TRACE("Sent {} bytes via ZeroMQ", data.size());
151 return true;
152
153 } catch (const zmq::error_t& e) {
154 LOG_ERROR("ZeroMQ send failed: {}", e.what());
155 return false;
156 }

159std::optional<std::vector<uint8_t>> ZeroMQSocket::receive(int32_t timeoutMs) {
160 if (!connected_.load()) {
161 LOG_WARN("Cannot receive - socket not connected");
162 return std::nullopt;
163 }
164
165 std::lock_guard<std::mutex> lock(mutex_);
166
167 if (!socket_) {
168 LOG_WARN("Cannot receive - socket is null");
169 return std::nullopt;
170 }
171
172 try {
173 // Set receive timeout
174 socket_->set(zmq::sockopt::rcvtimeo, timeoutMs);
175
176 zmq::message_t message;
177 auto result = socket_->recv(message, zmq::recv_flags::none);
178
179 if (!result.has_value()) {
180 // Timeout - normal condition, don't log
181 return std::nullopt;
182 }
183
184 // Copy message data to vector
185 std::vector<uint8_t> data(message.size());
186 std::memcpy(data.data(), message.data(), message.size());
187
188 LOG_TRACE("Received {} bytes via ZeroMQ", data.size());
189 return data;
190
191 } catch (const zmq::error_t& e) {
192 if (e.num() == EAGAIN || e.num() == ETIMEDOUT) {
193 // Timeout - normal condition
194 return std::nullopt;
195 }
196 LOG_ERROR("ZeroMQ receive failed: {}", e.what());
197 return std::nullopt;
