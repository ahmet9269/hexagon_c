201void TrackDataZeroMQIncomingAdapter::receiveLoop() {
202 LOG_DEBUG("Receive loop started");
203
204 while (!stopRequested_.load()) {
205 try {
206 zmq::message_t message;
207
208 // Non-blocking receive with timeout
209 auto result = socket_->recv(message, zmq::recv_flags::none);
210
211 if (result.has_value() && result.value() > 0) {
212 // Record receive timestamp for latency calculation
213 auto receive_time = std::chrono::duration_cast<std::chrono::microseconds>(
214 std::chrono::high_resolution_clock::now().time_since_epoch()).count();
215
216 // Convert message to byte vector
217 std::vector<uint8_t> data(
218 static_cast<const uint8_t*>(message.data()),
219 static_cast<const uint8_t*>(message.data()) + message.size()
220 );
221
222 // Deserialize using model's built-in deserialization
223 domain::model::TrackData trackData;
224 if (trackData.deserialize(data)) {
225 if (trackData.isValid() && incomingPort_) {
226 // Calculate latency
227 auto latency_us = receive_time - trackData.getOriginalUpdateTime();
228
229 // Log latency metrics (async, ~20ns overhead)
230 utils::Logger::logTrackReceived(trackData.getTrackId(), latency_us);
231
232 // Forward to domain service via hexagonal architecture port
233 incomingPort_->processAndForwardTrackData(trackData);
234
235 LOG_TRACE("Processed TrackData - ID: {}", trackData.getTrackId());
236 } else {
237 LOG_WARN("Invalid TrackData received or incoming port null");
238 }
239 } else {
240 LOG_WARN("Failed to deserialize TrackData message ({} bytes)", data.size());
241 }
242 }
243 // Timeout is normal, just continue loop
244
245 } catch (const zmq::error_t& e) {
246 if (!stopRequested_.load()) {
247 LOG_ERROR("ZeroMQ receive error: {}", e.what());
248 }
249 // Brief sleep on error before retry
250 std::this_thread::sleep_for(std::chrono::microseconds(100));
251 } catch (const std::exception& e) {
252 LOG_ERROR("Worker thread error: {}", e.what());
253 std::this_thread::sleep_for(std::chrono::microseconds(100));
254 }
255 }
256
257 LOG_DEBUG("Receive loop ended");
258}

260bool TrackDataZeroMQIncomingAdapter::deserializeTrackData(
261 const std::vector<uint8_t>& data,
262 domain::model::TrackData& trackData) {
263
264 // Binary deserialization format:
265 // [TrackId: int32_t]
266 // [XVelocityECEF: double]
267 // [YVelocityECEF: double]
268 // [ZVelocityECEF: double]
269 // [XPositionECEF: double]
270 // [YPositionECEF: double]
271 // [ZPositionECEF: double]
272 // [OriginalUpdateTime: int64_t]
273
274 constexpr size_t EXPECTED_SIZE = sizeof(int32_t) +
275 (6 * sizeof(double)) +
276 sizeof(int64_t);
277
278 if (data.size() < EXPECTED_SIZE) {
279 LOG_DEBUG("Invalid data size: {} (expected {})", data.size(), EXPECTED_SIZE);
280 return false;
281 }
282
283 try {
284 size_t offset = 0;
285
286 // TrackId
287 int32_t trackId;
288 std::memcpy(&trackId, data.data() + offset, sizeof(int32_t));
289 offset += sizeof(int32_t);
290 trackData.setTrackId(trackId);
291
292 // Velocities
293 double xVel, yVel, zVel;
294 std::memcpy(&xVel, data.data() + offset, sizeof(double));
295 offset += sizeof(double);
296 std::memcpy(&yVel, data.data() + offset, sizeof(double));
297 offset += sizeof(double);
298 std::memcpy(&zVel, data.data() + offset, sizeof(double));
299 offset += sizeof(double);
300 trackData.setXVelocityECEF(xVel);
301 trackData.setYVelocityECEF(yVel);
302 trackData.setZVelocityECEF(zVel);
303
304 // Positions
305 double xPos, yPos, zPos;
306 std::memcpy(&xPos, data.data() + offset, sizeof(double));
307 offset += sizeof(double);
308 std::memcpy(&yPos, data.data() + offset, sizeof(double));
309 offset += sizeof(double);
310 std::memcpy(&zPos, data.data() + offset, sizeof(double));
311 offset += sizeof(double);
312 trackData.setXPositionECEF(xPos);
313 trackData.setYPositionECEF(yPos);
314 trackData.setZPositionECEF(zPos);
315
316 // Timestamp
317 int64_t timestamp;
318 std::memcpy(&timestamp, data.data() + offset, sizeof(int64_t));
319 trackData.setOriginalUpdateTime(static_cast<long>(timestamp));
320
321 return true;
322
323 } catch (const std::exception& e) {
324 LOG_ERROR("Deserialization error: {}", e.what());
325 return false;
326 }
