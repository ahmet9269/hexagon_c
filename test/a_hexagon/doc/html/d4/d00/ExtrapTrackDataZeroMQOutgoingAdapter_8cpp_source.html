218 {

235void ExtrapTrackDataZeroMQOutgoingAdapter::publisherWorker() {
236 LOG_DEBUG("Publisher worker started: ExtrapTrackDataAdapter");
237
238 while (running_.load()) {
239 domain::model::ExtrapTrackData data;
240
241 // Wait for message with timeout
242 {
243 std::unique_lock<std::mutex> lock(queueMutex_);
244
245 if (queueCV_.wait_for(lock, std::chrono::milliseconds(100),
246 [this]() { return !messageQueue_.empty() || !running_.load(); })) {
247
248 if (!running_.load()) {
249 break;
250 }
251
252 if (!messageQueue_.empty()) {
253 data = std::move(messageQueue_.front());
254 messageQueue_.pop();
255 } else {
256 continue;
257 }
258 } else {
259 continue; // Timeout, check running flag
260 }
261 }
262
263 // Serialize and send (outside lock)
264 try {
265 std::vector<uint8_t> binaryData = data.serialize();
266
267 if (binaryData.empty()) {
268 LOG_ERROR("Empty payload for track ID: {}", data.getTrackId());
269 continue;
270 }
271
272 // Send via IMessageSocket abstraction
273 if (socket_->send(binaryData, group_)) {
274 LOG_DEBUG("[a_hexagon] ExtrapTrackData sent - TrackID: {}, Size: {} bytes",
275 data.getTrackId(), binaryData.size());
276 } else {
277 LOG_WARN("Failed to send ExtrapTrackData - TrackID: {}", data.getTrackId());
278 }
279
280 } catch (const std::exception& e) {
281 LOG_ERROR("Failed to send message: {}", e.what());
282 }
283 }
284
285 LOG_DEBUG("Publisher worker stopped: ExtrapTrackDataAdapter");
286}
